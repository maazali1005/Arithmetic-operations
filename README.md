
# ARITHMETIC-OPERATIONS
import numpy as np

arr1=np.array([1,2,3,4])

arr2=np.array([5,6,7,8])

#Addition
print("addition:",np.add(arr1,arr2))

#Subtraction

print("Subtraction:",np.subtract(arr1,arr2))

#Multiplication

print("multiplication:",np.multiply(arr1,arr2))
